    ```tsx
    /**
     * @description Per-user admin detail page with Profile + Permissions tabs. Deep-links via ?tab= query param.
     * @returns {JSX.Element} Detail page with tabbed content
     */
    export default function UserDetailPage() {
      const { id } = useParams<{ id: string }>()
      const [searchParams, setSearchParams] = useSearchParams()
      const activeTab = searchParams.get('tab') ?? 'profile'
      const userId = Number(id)

      // Fetch user via existing user-by-id hook (create one if missing: useUser(userId) in userQueries.ts)
      const { data: user, isLoading } = useUser(userId)

      if (isLoading) return <LoadingSkeleton />
      if (!user) return <div>{t('users.notFound')}</div>

      return (
        <div className="p-6">
          <h1 className="text-2xl font-bold mb-4 dark:text-white">{user.display_name}</h1>
          <Tabs value={activeTab} onValueChange={(v) => setSearchParams({ tab: v })}>
            <TabsList>
              <TabsTrigger value="profile">{t('users.detail.tabs.profile')}</TabsTrigger>
              <TabsTrigger value="permissions">{t('users.detail.tabs.permissions')}</TabsTrigger>
            </TabsList>
            <TabsContent value="profile"><UserProfileTab user={user} /></TabsContent>
            <TabsContent value="permissions"><OverrideEditor userId={userId} userRole={user.role} /></TabsContent>
          </Tabs>
        </div>
      )
    }
    ```
