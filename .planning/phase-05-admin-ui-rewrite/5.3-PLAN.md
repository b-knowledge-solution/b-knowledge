    ```ts
    const PRINCIPAL_TYPE_USER = 'user' as const
    const PRINCIPAL_TYPE_TEAM = 'team' as const
    const PRINCIPAL_TYPE_ROLE = 'role' as const
    type PrincipalType = typeof PRINCIPAL_TYPE_USER | typeof PRINCIPAL_TYPE_TEAM | typeof PRINCIPAL_TYPE_ROLE

    const FILTER_ALL = 'all' as const
    type FilterChip = typeof FILTER_ALL | PrincipalType
    ```

    Component:
    ```tsx
    interface Principal { type: PrincipalType; id: number | string; label: string; sublabel?: string }
    interface Props { onSelect: (p: Principal) => void; excludeIds?: string[] }

    export function PrincipalPicker({ onSelect, excludeIds = [] }: Props) {
      const { data: users = [] } = useUsers() // or equivalent existing query
      const { data: teams = [] } = useTeams()
      const roles = ROLES // from @/constants/roles
      const [filter, setFilter] = useState<FilterChip>(FILTER_ALL)
      const [query, setQuery] = useState('')

      const merged: Principal[] = [
        ...users.map(u => ({ type: PRINCIPAL_TYPE_USER, id: u.id, label: u.display_name, sublabel: u.email })),
        ...teams.map(t => ({ type: PRINCIPAL_TYPE_TEAM, id: t.id, label: t.name })),
        ...roles.map(r => ({ type: PRINCIPAL_TYPE_ROLE, id: r, label: r })),
      ]

      const filtered = merged
        .filter(p => !excludeIds.includes(`${p.type}:${p.id}`))
        .filter(p => filter === FILTER_ALL || p.type === filter)
        .filter(p => p.label.toLowerCase().includes(query.toLowerCase()))

      return (
        <Command>
          <div className="flex gap-2 p-2">
            {[FILTER_ALL, PRINCIPAL_TYPE_USER, PRINCIPAL_TYPE_TEAM, PRINCIPAL_TYPE_ROLE].map(chip => (
              <button key={chip} onClick={() => setFilter(chip)} className={cn('chip', filter === chip && 'active', 'dark:bg-slate-700')}>
                {t(`permissions.admin.picker.chip.${chip}`)}
              </button>
            ))}
          </div>
          <CommandInput value={query} onValueChange={setQuery} placeholder={t('permissions.admin.picker.searchPlaceholder')} />
          <CommandList>
            {filtered.map(p => (
              <CommandItem key={`${p.type}:${p.id}`} onSelect={() => onSelect(p)}>
                <Badge>{t(`permissions.admin.picker.type.${p.type}`)}</Badge>
                <span>{p.label}</span>
                {p.sublabel && <span className="text-xs text-muted-foreground">{p.sublabel}</span>}
              </CommandItem>
            ))}
          </CommandList>
        </Command>
      )
    }
    ```

    Export as named `PrincipalPicker`. Full JSDoc. Dark-mode classes.
