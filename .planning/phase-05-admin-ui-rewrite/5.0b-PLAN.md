    **Step 2: Raw HTTP client (`fe/src/features/permissions/api/permissionsApi.ts`)**

    Define a local constant `PERMISSIONS_API_BASE = '/api/permissions'` at module top (no hardcoded path literals inside functions per CLAUDE.md rule). Export these functions with full JSDoc:

    ```ts
    /**
     * @description Fetch the full permission catalog from the backend (normally uses snapshot in Phase 4, available for runtime refresh).
     */
    export async function getCatalog(): Promise<{ keys: string[]; version: string }> {
      return api.get(`${PERMISSIONS_API_BASE}/catalog`)
    }

    /** @description List users in the caller's tenant who effectively have the given action on the given subject. */
    export async function whoCanDo(action: string, subject: string, resourceId?: string): Promise<WhoCanDoResult> {
      const params = new URLSearchParams({ action, subject })
      if (resourceId) params.set('resource_id', resourceId)
      return api.get(`${PERMISSIONS_API_BASE}/who-can-do?${params.toString()}`)
    }

    export async function getRolePermissions(role: string): Promise<RolePermissions> { return api.get(`${PERMISSIONS_API_BASE}/roles/${encodeURIComponent(role)}`) }

    export async function updateRolePermissions(role: string, body: UpdateRolePermissionsBody): Promise<RolePermissions> { return api.put(`${PERMISSIONS_API_BASE}/roles/${encodeURIComponent(role)}`, body) }

    export async function getUserOverrides(userId: number): Promise<UserPermissionOverride[]> { return api.get(`${PERMISSIONS_API_BASE}/users/${userId}/overrides`) }

    export async function createOverride(userId: number, body: CreateOverrideBody): Promise<UserPermissionOverride> { return api.post(`${PERMISSIONS_API_BASE}/users/${userId}/overrides`, body) }

    /** @description Delete an override by its id. NOTE: the route is flat (/overrides/:id), NOT nested under /users/:userId. */
    export async function deleteOverride(id: number): Promise<void> { return api.delete(`${PERMISSIONS_API_BASE}/overrides/${id}`) }

    export async function getGrants(resourceType: GrantResourceType, resourceId: string): Promise<ResourceGrant[]> {
      const params = new URLSearchParams({ resource_type: resourceType, resource_id: resourceId })
      return api.get(`${PERMISSIONS_API_BASE}/grants?${params.toString()}`)
    }

    export async function createGrant(body: CreateGrantBody): Promise<ResourceGrant> { return api.post(`${PERMISSIONS_API_BASE}/grants`, body) }

    export async function deleteGrant(id: number): Promise<void> { return api.delete(`${PERMISSIONS_API_BASE}/grants/${id}`) }
    ```

    ```ts
    export function useRolePermissions(role: string) {
      return useQuery({ queryKey: queryKeys.permissions.rolePermissions(role), queryFn: () => permissionsApi.getRolePermissions(role) })
    }

    export function useUpdateRolePermissions() {
      const qc = useQueryClient()
      return useMutation({
        mutationFn: ({ role, permission_keys }: { role: string; permission_keys: string[] }) =>
          permissionsApi.updateRolePermissions(role, { permission_keys }),
        onSuccess: (_, { role }) => {
          qc.invalidateQueries({ queryKey: queryKeys.permissions.rolePermissions(role) })
        },
      })
    }
    ```
