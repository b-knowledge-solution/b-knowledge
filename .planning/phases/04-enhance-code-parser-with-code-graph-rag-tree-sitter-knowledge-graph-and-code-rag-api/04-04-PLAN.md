2. Create `be/src/shared/services/memgraphService.ts`:

   ```typescript
   import neo4j, { Driver, Session, Record as Neo4jRecord } from 'neo4j-driver'
   import { config } from '../config'
   
   /**
    * @description Singleton service for Memgraph graph database access via Bolt protocol.
    * Provides Cypher query execution with session management.
    */
   class MemgraphService {
     private driver: Driver | null = null
     
     private getDriver(): Driver {
       if (!this.driver) {
         this.driver = neo4j.driver(config.memgraph.boltUrl)
       }
       return this.driver
     }
     
     async query<T = Record<string, unknown>>(
       cypher: string, 
       params: Record<string, unknown> = {}
     ): Promise<T[]> {
       const session = this.getDriver().session()
       try {
         const result = await session.run(cypher, params)
         return result.records.map(r => r.toObject() as T)
       } finally {
         await session.close()
       }
     }
     
     async close(): Promise<void> {
       if (this.driver) {
         await this.driver.close()
         this.driver = null
       }
     }
   }
   
   export const memgraphService = new MemgraphService()
   ```

3. Add `memgraph` section to config.ts:
   ```typescript
   memgraph: {
     boltUrl: process.env.MEMGRAPH_BOLT_URL || 'bolt://localhost:7687',
   }
   ```

```typescript
import { memgraphService } from '../../shared/services/memgraphService'

/**
 * @description Service for querying code knowledge graph in Memgraph.
 * Provides structured Cypher queries for code structure analysis.
 */
class CodeGraphService {
  
  async getStats(kbId: string) {
    const nodes = await memgraphService.query(
      'MATCH (n {kb_id: $kbId}) RETURN labels(n) AS label, count(n) AS count',
      { kbId }
    )
    const rels = await memgraphService.query(
      'MATCH (n {kb_id: $kbId})-[r]->(m {kb_id: $kbId}) RETURN type(r) AS type, count(r) AS count',
      { kbId }
    )
    return { nodes, relationships: rels }
  }
  
  async getCallers(kbId: string, functionName: string) {
    return memgraphService.query(
      `MATCH (caller {kb_id: $kbId})-[:CALLS]->(target {kb_id: $kbId})
       WHERE target.name = $name OR target.qualified_name ENDS WITH $dotName
       RETURN caller.qualified_name AS caller, caller.path AS file, 
              target.qualified_name AS target`,
      { kbId, name: functionName, dotName: '.' + functionName }
    )
  }
  
  async getCallees(kbId: string, functionName: string) {
    return memgraphService.query(
      `MATCH (source {kb_id: $kbId})-[:CALLS]->(callee {kb_id: $kbId})
       WHERE source.name = $name OR source.qualified_name ENDS WITH $dotName
       RETURN callee.qualified_name AS callee, callee.path AS file,
              source.qualified_name AS source`,
      { kbId, name: functionName, dotName: '.' + functionName }
    )
  }
  
  async getHierarchy(kbId: string, className: string) {
    return memgraphService.query(
      `MATCH path = (child {kb_id: $kbId})-[:INHERITS*1..5]->(parent {kb_id: $kbId})
       WHERE child.name = $name OR parent.name = $name
       RETURN [n IN nodes(path) | {name: n.name, qualified_name: n.qualified_name, 
              labels: labels(n)}] AS chain`,
      { kbId, name: className }
    )
  }
  
  async getSnippet(kbId: string, functionName: string) {
    return memgraphService.query(
      `MATCH (n {kb_id: $kbId})
       WHERE (n:Function OR n:Method) AND 
             (n.name = $name OR n.qualified_name ENDS WITH $dotName)
       RETURN n.qualified_name AS name, n.source_code AS code,
              n.path AS file, n.start_line AS startLine, n.end_line AS endLine`,
      { kbId, name: functionName, dotName: '.' + functionName }
    )
  }
  
  async getFullGraph(kbId: string, limit: number = 500) {
    const nodes = await memgraphService.query(
      'MATCH (n {kb_id: $kbId}) RETURN id(n) AS id, labels(n) AS labels, properties(n) AS props LIMIT $limit',
      { kbId, limit: neo4j.int(limit) }
    )
    const edges = await memgraphService.query(
      `MATCH (n {kb_id: $kbId})-[r]->(m {kb_id: $kbId}) 
       RETURN id(n) AS source, id(m) AS target, type(r) AS type LIMIT $limit`,
      { kbId, limit: neo4j.int(limit) }
    )
    return { nodes, edges }
  }
}

export const codeGraphService = new CodeGraphService()
```

```typescript
async queryWithAI(kbId: string, question: string, tenantId: string) {
  // 1. Get graph schema
  const schema = await this.getStats(kbId)
  
  // 2. Build prompt for NL → Cypher
  const systemPrompt = `You are a Cypher query generator for a code knowledge graph.
Graph schema:
- Node labels: ${JSON.stringify(schema.nodes.map(n => n.label))}
- Relationship types: ${JSON.stringify(schema.relationships.map(r => r.type))}
- Key properties: qualified_name, name, path, source_code, start_line, end_line, kb_id

Rules:
- Always filter with {kb_id: '${kbId}'}
- Return qualified_name, path, and relevant properties
- Use LIMIT 50 unless user specifies otherwise
- Return ONLY the Cypher query, no explanation`
  
  // 3. Call tenant's LLM to generate Cypher
  const cypher = await llmService.chat(tenantId, systemPrompt, question)
  
  // 4. Execute generated Cypher
  const results = await memgraphService.query(cypher, { kbId })
  
  return { cypher, results }
}
```
