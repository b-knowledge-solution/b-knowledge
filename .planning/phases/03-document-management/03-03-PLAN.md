export async function bulkUpdateMetadata(datasetIds: string[], metadata: Record<string, string>, mode: 'merge' | 'overwrite'): Promise<void> {

export async function getTagAggregations(datasetIds?: string[]): Promise<{ key: string; values: string[] }[]> {
  const params = datasetIds ? `?dataset_ids=${datasetIds.join(',')}` : ''
  return api.get(`/api/rag/tags/aggregations${params}`)
}

**6. Add query hooks** in `datasetQueries.ts`:
```typescript
export function useBulkUpdateMetadata() {
  return useMutation({
    mutationFn: (params: { datasetIds: string[]; metadata: Record<string, string>; mode: 'merge' | 'overwrite' }) =>
      bulkUpdateMetadata(params.datasetIds, params.metadata, params.mode),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.datasets.all })
    }
  })
}

export function useTagAggregations(datasetIds?: string[]) {
  return useQuery({
    queryKey: ['tags', 'aggregations', datasetIds],
    queryFn: () => getTagAggregations(datasetIds),
  })
}
```
