```typescript
/** @description Upload a new version of an existing document as a new dataset */
export async function createDatasetVersion(
  datasetId: string,
  files: File[],
  changeSummary?: string,
  autoParse?: boolean
): Promise<Dataset> {
  const formData = new FormData()
  files.forEach(f => formData.append('files', f))
  if (changeSummary) formData.append('change_summary', changeSummary)
  if (autoParse) formData.append('auto_parse', 'true')
  return api.post(`/api/rag/datasets/${datasetId}/versions`, formData)
}
```
