Create provider that fetches abilities from backend API:
```typescript
export function AbilityProvider({ children }: { children: React.ReactNode }) {
  const [ability, setAbility] = useState<AppAbility>(defaultAbility)

  useEffect(() => {
    // Fetch user abilities from backend
    async function loadAbilities() {
      try {
        const res = await fetch('/api/auth/abilities', { credentials: 'include' })
        if (res.ok) {
          const data = await res.json()
          setAbility(createMongoAbility<[Actions, Subjects]>(data.rules))
        }
      } catch (err) {
        // On error, keep default (no permissions) -- user will see access denied
        console.error('Failed to load abilities', err)
      }
    }
    loadAbilities()
  }, [])

  return (
    <AbilityContext.Provider value={ability}>
      {children}
    </AbilityContext.Provider>
  )
}
```

Component:
```typescript
interface RoleBadgeProps { role: string; className?: string }

export function RoleBadge({ role, className }: RoleBadgeProps) {
  const { t } = useTranslation()
  const roleKey = role.toLowerCase()
  const style = ROLE_STYLES[roleKey] || ROLE_STYLES['user']
  const label = t(`accessControl.roles.${roleKey === 'super-admin' ? 'superAdmin' : roleKey}`)
  return (
    <Badge variant="secondary" className={`${style} ${className || ''}`} role="status" aria-label={`Role: ${label}`}>
      {label}
    </Badge>
  )
}
```

**Step 1: Create OrgSwitcher component**

Create `fe/src/components/OrgSwitcher.tsx`:

```typescript
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select'
import { useTranslation } from 'react-i18next'
```

Interface:
```typescript
interface Org { id: string; name: string; role: string }
interface OrgSwitcherProps { orgs: Org[]; currentOrgId: string; onSwitch: (orgId: string) => void; isLoading?: boolean }
```

Component logic:
- If `orgs.length <= 1`: render static text showing org name (no dropdown)
- If `orgs.length > 1`: render Select dropdown with org names and role labels in parentheses
- On change: call `onSwitch(orgId)` which triggers API call to `POST /api/auth/switch-org` and page reload
- During switch: show spinner overlay with `t('accessControl.orgSwitcher.switching')` text
- `aria-label="Switch organization"`

```typescript
const ability = useAppAbility()
```
