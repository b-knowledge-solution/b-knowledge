
- Add unique constraint on (user_id, tenant_id) to prevent duplicate memberships:
```sql
ALTER TABLE user_tenant ADD CONSTRAINT uq_user_tenant UNIQUE (user_id, tenant_id);
```

3. **Add ABAC policy columns to knowledgebase:**
```sql
ALTER TABLE knowledgebase ADD COLUMN IF NOT EXISTS policy_rules JSONB DEFAULT '[]';
```

4. **Add document-level permission overrides:**
```sql
ALTER TABLE document ADD COLUMN IF NOT EXISTS policy_overrides JSONB DEFAULT '[]';
```

5. **Create platform_policies table:**
```sql
CREATE TABLE platform_policies (
  id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
  name TEXT NOT NULL,
  description TEXT,
  rules JSONB NOT NULL DEFAULT '[]',
  is_active BOOLEAN DEFAULT true,
  created_by TEXT,
  updated_by TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

6. **Add tenant_id to audit_logs:**
```sql
ALTER TABLE audit_logs ADD COLUMN IF NOT EXISTS tenant_id TEXT;
CREATE INDEX IF NOT EXISTS idx_audit_logs_tenant_id ON audit_logs(tenant_id);
```
