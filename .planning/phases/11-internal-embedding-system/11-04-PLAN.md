 * @returns {Promise<{ taskId: string }>} The created task ID

**2. Add re-embed mutation hook to `fe/src/features/datasets/api/datasetsQueries.ts`:**

```typescript
/**
 * @description Mutation hook to trigger re-embedding of all chunks in a dataset.
 * Shows success toast on completion and error toast on failure.
 */
export function useReEmbedDataset() {
    const { t } = useTranslation()
    return useMutation({
        mutationFn: (datasetId: string) => datasetsApi.reEmbedDataset(datasetId),
        onSuccess: () => {
            toast.success(t('datasetSettings.reembed.successToast'))
        },
        onError: () => {
            toast.error(t('datasetSettings.reembed.errorToast'))
        },
    })
}
```
