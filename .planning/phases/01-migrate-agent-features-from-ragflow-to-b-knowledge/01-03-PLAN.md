**agent.schemas.ts — Zod validation schemas:**

```typescript
import { z } from 'zod'

export const createAgentSchema = z.object({
  name: z.string().min(1).max(255),
  description: z.string().max(2000).optional(),
  mode: z.enum(['agent', 'pipeline']).default('agent'),
  project_id: z.string().uuid().optional(),
  template_id: z.string().uuid().optional(),
})

export const updateAgentSchema = z.object({
  name: z.string().min(1).max(255).optional(),
  description: z.string().max(2000).nullable().optional(),
  status: z.enum(['draft', 'published']).optional(),
  dsl: z.record(z.unknown()).optional(),
})

export const saveVersionSchema = z.object({
  version_label: z.string().max(128).optional(),
  change_summary: z.string().max(1000).optional(),
})

export const agentIdParamSchema = z.object({
  id: z.string().uuid(),
})

export const versionIdParamSchema = z.object({
  id: z.string().uuid(),
  versionId: z.string().uuid(),
})

export const listAgentsQuerySchema = z.object({
  mode: z.enum(['agent', 'pipeline']).optional(),
  status: z.enum(['draft', 'published']).optional(),
  project_id: z.string().uuid().optional(),
  page: z.coerce.number().int().positive().default(1),
  page_size: z.coerce.number().int().min(1).max(100).default(20),
  search: z.string().max(255).optional(),
})
```

**agent.service.ts — Singleton service with CRUD + versioning:**

Implement as a class with singleton export pattern (matching existing services like projects.service.ts).

Methods:
- `list(tenantId: string, filters: ListAgentsQuery)` — Paginated list with optional mode/status/project_id/search filters. Only returns root agents (parent_id IS NULL). Uses Knex `.where()` chaining.
- `getById(id: string, tenantId: string)` — Fetch by ID with tenant guard. Throws 404 if not found or wrong tenant.
- `create(data: CreateAgentDto, tenantId: string, userId: string)` — Create agent. If template_id provided, copy DSL from template. Sets created_by, tenant_id.
- `update(id: string, data: UpdateAgentDto, tenantId: string)` — Update agent fields. If dsl is updated, only update if status is 'draft' (published agents are immutable).
- `delete(id: string, tenantId: string)` — Soft check: delete agent and all its versions (CASCADE handles runs/steps).
- `duplicate(id: string, tenantId: string, userId: string)` — Clone agent with name "{name} (copy)", reset to draft.
- `saveVersion(id: string, tenantId: string, userId: string, label?: string)` — Version-as-row: create new row with parent_id = id, version_number = max existing + 1, copy current DSL snapshot.
- `listVersions(id: string, tenantId: string)` — List all versions of an agent (where parent_id = id), ordered by version_number DESC.
- `restoreVersion(id: string, versionId: string, tenantId: string)` — Copy version's DSL back to parent agent, mark parent as draft.
- `deleteVersion(versionId: string, tenantId: string)` — Delete a specific version row. Prevent deleting if it's the parent.
- `exportJson(id: string, tenantId: string)` — Return agent with DSL for JSON export.

All methods must:
- Include JSDoc with @description, @param, @returns, @throws
- Use ModelFactory.agents for DB access
- Include tenant_id checks on every query
- Log operations via logger

**agent.routes.ts — Express router:**

```typescript
const router = Router()

// All routes require authentication and tenant context
router.use(requireAuth)
router.use(requireTenant)

// CRUD
router.get('/', validate({ query: listAgentsQuerySchema }), agentController.listAgents)
router.post('/', validate(createAgentSchema), agentController.createAgent)
router.get('/:id', validate({ params: agentIdParamSchema }), agentController.getAgent)
router.put('/:id', validate({ params: agentIdParamSchema, body: updateAgentSchema }), agentController.updateAgent)
router.delete('/:id', validate({ params: agentIdParamSchema }), agentController.deleteAgent)

// Actions
router.post('/:id/duplicate', validate({ params: agentIdParamSchema }), agentController.duplicateAgent)
router.get('/:id/export', validate({ params: agentIdParamSchema }), agentController.exportAgent)

// Versioning
router.get('/:id/versions', validate({ params: agentIdParamSchema }), agentController.listVersions)
router.post('/:id/versions', validate({ params: agentIdParamSchema, body: saveVersionSchema }), agentController.saveVersion)
router.post('/:id/versions/:versionId/restore', validate({ params: versionIdParamSchema }), agentController.restoreVersion)
router.delete('/:id/versions/:versionId', validate({ params: versionIdParamSchema }), agentController.deleteVersion)
```
