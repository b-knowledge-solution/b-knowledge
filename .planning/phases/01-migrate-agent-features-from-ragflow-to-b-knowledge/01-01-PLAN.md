**Migration 1: 20260323000000_create_agents.ts**
```sql
CREATE TABLE agents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(255) NOT NULL,
  description TEXT,
  avatar VARCHAR(512),
  mode VARCHAR(20) NOT NULL DEFAULT 'agent',  -- 'agent' | 'pipeline'
  status VARCHAR(20) NOT NULL DEFAULT 'draft', -- 'draft' | 'published'
  dsl JSONB NOT NULL DEFAULT '{}',
  dsl_version INTEGER NOT NULL DEFAULT 1,
  policy_rules JSONB,                          -- ABAC policies (matches dataset pattern)
  tenant_id VARCHAR(64) NOT NULL,
  project_id UUID REFERENCES projects(id) ON DELETE SET NULL,
  parent_id UUID REFERENCES agents(id) ON DELETE SET NULL,
  version_number INTEGER NOT NULL DEFAULT 0,
  version_label VARCHAR(128),
  created_by TEXT REFERENCES users(id) ON DELETE SET NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_agents_tenant_parent ON agents(tenant_id, parent_id);
CREATE INDEX idx_agents_tenant_status ON agents(tenant_id, status);
CREATE INDEX idx_agents_created_by ON agents(created_by);
CREATE INDEX idx_agents_project_id ON agents(project_id);
```

**Migration 2: 20260323000001_create_agent_runs.ts**
```sql
CREATE TABLE agent_runs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id UUID NOT NULL REFERENCES agents(id) ON DELETE CASCADE,
  tenant_id VARCHAR(64) NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'pending', -- 'pending' | 'running' | 'completed' | 'failed' | 'cancelled'
  mode VARCHAR(20) NOT NULL,                      -- 'agent' | 'pipeline'
  input TEXT,
  output TEXT,
  error TEXT,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  duration_ms INTEGER,
  total_nodes INTEGER DEFAULT 0,
  completed_nodes INTEGER DEFAULT 0,
  triggered_by TEXT REFERENCES users(id) ON DELETE SET NULL,
  trigger_type VARCHAR(20) NOT NULL DEFAULT 'manual', -- 'manual' | 'webhook' | 'embed'
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_agent_runs_agent_id ON agent_runs(agent_id);
CREATE INDEX idx_agent_runs_tenant_status ON agent_runs(tenant_id, status);

CREATE TABLE agent_run_steps (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID NOT NULL REFERENCES agent_runs(id) ON DELETE CASCADE,
  node_id VARCHAR(255) NOT NULL,
  node_type VARCHAR(100) NOT NULL,
  node_label VARCHAR(255),
  status VARCHAR(20) NOT NULL DEFAULT 'pending', -- 'pending' | 'running' | 'completed' | 'failed' | 'skipped'
  input_data JSONB,
  output_data JSONB,
  error TEXT,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  duration_ms INTEGER,
  execution_order INTEGER NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_agent_run_steps_run_id ON agent_run_steps(run_id);
CREATE INDEX idx_agent_run_steps_run_node ON agent_run_steps(run_id, node_id);
```

**Migration 3: 20260323000002_create_agent_tool_credentials.ts**
```sql
CREATE TABLE agent_tool_credentials (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id VARCHAR(64) NOT NULL,
  agent_id UUID REFERENCES agents(id) ON DELETE CASCADE,  -- NULL = tenant-level default
  tool_type VARCHAR(100) NOT NULL,                         -- e.g., 'tavily', 'github', 'sql'
  name VARCHAR(255) NOT NULL,
  encrypted_credentials TEXT NOT NULL,                     -- AES-256-CBC via crypto.service.ts
  created_by TEXT REFERENCES users(id) ON DELETE SET NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_agent_tool_creds_tenant ON agent_tool_credentials(tenant_id);
CREATE INDEX idx_agent_tool_creds_agent ON agent_tool_credentials(agent_id);
CREATE UNIQUE INDEX idx_agent_tool_creds_unique ON agent_tool_credentials(tenant_id, COALESCE(agent_id, '00000000-0000-0000-0000-000000000000'), tool_type);
```
