       ```typescript
       export class BudgetTracker {
         private tokensUsed = 0
         private callsUsed = 0
         constructor(
           private readonly maxTokens: number = 50_000,
           private readonly maxCalls: number = 15
         ) {}
         recordCall(tokens: number): void { this.tokensUsed += tokens; this.callsUsed += 1 }
         isExhausted(): boolean { return this.tokensUsed >= this.maxTokens || this.callsUsed >= this.maxCalls }
         getStatus() { return { tokensUsed: this.tokensUsed, tokensMax: this.maxTokens, callsUsed: this.callsUsed, callsMax: this.maxCalls } }
       }
       ```

       export interface DeepResearchProgressEvent {

         query?: string
         depth?: number
         index?: number
         total?: number
         chunks?: number
         message?: string
         tokensUsed?: number
         tokensMax?: number
         callsUsed?: number
         callsMax?: number
         completed?: number
       }
