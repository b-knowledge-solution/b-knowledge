    ```ts
    /**
     * @description Resolve a set of document category IDs to the flat, deduped
     * list of underlying dataset IDs. Used by Phase 6 grant enforcement
     * (resolveGrantedDatasetsForUser) to translate DocumentCategory grants into
     * RAG-search dataset IDs.
     *
     * Two data sources are UNIONed:
     *   1. `document_categories.dataset_id` — single nullable column, populated
     *      for `category_type IN ('standard', 'code')`.
     *   2. `document_category_versions.ragflow_dataset_id` — one row per version,
     *      populated for `category_type='documents'`. Only versions whose
     *      status is in SEARCHABLE_VERSION_STATUSES (ACTIVE or SYNCING) are
     *      considered — archived versions and any future non-whitelisted
     *      status are hidden from grant-based search. See Phase 6 RESEARCH.md
     *      §11 R-J for the whitelist rationale.
     *
     * NULL dataset_ids are filtered out in SQL so callers never see them.
     *
     * @param {readonly string[]} categoryIds - Category IDs to resolve.
     * @returns {Promise<string[]>} Flat deduped dataset_id list. Empty array if input is empty.
     */
    async findDatasetIdsByCategoryIds(categoryIds: readonly string[]): Promise<string[]> {
      // Short-circuit on empty input — avoid emitting a `WHERE IN ()` which
      // some drivers treat as an always-false clause and others reject.
      if (categoryIds.length === 0) return []

      // Source 1: direct dataset_id on document_categories (standard/code types)
      const directRows = await this.knex(this.tableName)
        .whereIn('id', categoryIds as string[])
        .whereNotNull('dataset_id')
        .select('dataset_id')

      // Source 2: versioned dataset_ids on document_category_versions.
      // Whitelist by status — only versions that are ACTIVE or SYNCING expose
      // a searchable dataset. Archived (and any future non-whitelisted status)
      // are hidden from grant resolution. Uses SEARCHABLE_VERSION_STATUSES
      // constant per CLAUDE.md "No Hardcoded String Literals" rule.
      const versionRows = await this.knex('document_category_versions')
        .whereIn('category_id', categoryIds as string[])
        .whereNotNull('ragflow_dataset_id')
        .whereIn('status', SEARCHABLE_VERSION_STATUSES as unknown as string[])
        .select({ dataset_id: 'ragflow_dataset_id' })

      // Union + dedupe via Set — dataset_id is a string, set semantics are free.
      const all = new Set<string>()
      for (const r of directRows) if (r.dataset_id) all.add(r.dataset_id as string)
      for (const r of versionRows) if (r.dataset_id) all.add(r.dataset_id as string)
      return Array.from(all)
    }

    /**
     * @description Resolve a set of knowledge base IDs to the flat, deduped list
     * of underlying dataset IDs owned by their categories. Equivalent to
     * `findDatasetIdsByCategoryIds` but entered from the KB side — used when a
     * Phase 6 resource grant has `resource_type='knowledge_base'` and needs to
     * expand to every dataset the KB exposes.
     *
     * @param {readonly string[]} kbIds - Knowledge base IDs to resolve.
     * @returns {Promise<string[]>} Flat deduped dataset_id list.
     */
    async findDatasetIdsByKnowledgeBaseIds(kbIds: readonly string[]): Promise<string[]> {
      if (kbIds.length === 0) return []

      // Direct: categories whose knowledge_base_id is in the input and which
      // have a non-null dataset_id.
      const directRows = await this.knex(this.tableName)
        .whereIn('knowledge_base_id', kbIds as string[])
        .whereNotNull('dataset_id')
        .select('dataset_id')

      // Versioned: JOIN document_category_versions via category_id to reach
      // back to knowledge_base_id. Whitelist status via SEARCHABLE_VERSION_STATUSES
      // (active + syncing) — see R-J resolution in the JSDoc on findDatasetIdsByCategoryIds.
      const versionRows = await this.knex('document_category_versions as dcv')
        .join(`${this.tableName} as dc`, 'dcv.category_id', 'dc.id')
        .whereIn('dc.knowledge_base_id', kbIds as string[])
        .whereNotNull('dcv.ragflow_dataset_id')
        .whereIn('dcv.status', SEARCHABLE_VERSION_STATUSES as unknown as string[])
        .select({ dataset_id: 'dcv.ragflow_dataset_id' })

      const all = new Set<string>()
      for (const r of directRows) if (r.dataset_id) all.add(r.dataset_id as string)
      for (const r of versionRows) if (r.dataset_id) all.add(r.dataset_id as string)
      return Array.from(all)
    }
    ```

    ```ts
    import { ModelFactory } from '@/shared/models/index.js'
    import { ResourceType } from '@/shared/constants/resource-grants.js'
    import { log } from '@/shared/utils/logger.js'  // or whatever the existing logger import is

    /**
     * @description Soft cap on the number of grant-resolved dataset IDs a
     * single user can fan out to. OpenSearch's default `terms` query limit is
     * 65,536; we cap much lower to protect query latency. When exceeded, the
     * helper logs a warning and truncates (R-D in Phase 6 RESEARCH.md §11).
     */
    const GRANT_DATASET_SOFT_CAP = 10000

    /**
     * @description Resolve a user's active resource grants to the flat list of
     * dataset IDs they are permitted to search via Phase 6 Strategy A
     * (amendment A-2). Reuses `ResourceGrantModel.findActiveForUser` which
     * enforces `expires_at IS NULL OR expires_at > NOW()` in SQL (D-09) and
     * handles tenant isolation + team-membership disjunction. For each grant,
     * translates:
     *   - `resource_type='knowledge_base'` → every dataset owned by the KB
     *   - `resource_type='document_category'` → every dataset owned by the category
     * and returns the deduped union.
     *
     * Zero-grant users receive `[]` — callers unioning this into their kbIds
     * array are guaranteed a no-op, preserving parity (D-06).
     *
     * @param {string} userId - User whose grants to walk.
     * @param {string} tenantId - Tenant scope; cross-tenant grants excluded at the SQL layer.
     * @param {{ teamIds?: readonly string[] }} [opts] - Optional team membership for team-grant disjunction.
     * @returns {Promise<string[]>} Flat deduped dataset_id list (may be empty).
     */
    export async function resolveGrantedDatasetsForUser(
      userId: string,
      tenantId: string,
      opts: { teamIds?: readonly string[] } = {},
    ): Promise<string[]> {
      // Step 1: load active grants (SQL enforces expires_at + tenant + grantee disjunction)
      const grants = await ModelFactory.resourceGrant.findActiveForUser(
        userId,
        tenantId,
        opts.teamIds ?? [],
      )
      if (grants.length === 0) return []

      // Step 2: partition grant rows by resource_type so each resolver gets a
      // single batched call (avoid N+1).
      const kbIds: string[] = []
      const categoryIds: string[] = []
      for (const g of grants) {
        if (g.resource_type === ResourceType.KNOWLEDGE_BASE) kbIds.push(g.resource_id)
        else if (g.resource_type === ResourceType.DOCUMENT_CATEGORY) categoryIds.push(g.resource_id)
        // Any other resource_type is silently ignored — future resource kinds
        // will need their own branch here.
      }

      // Step 3: batch resolve both sides in parallel. Each resolver already
      // short-circuits on empty input.
      const [kbDatasets, categoryDatasets] = await Promise.all([
        ModelFactory.documentCategory.findDatasetIdsByKnowledgeBaseIds(kbIds),
        ModelFactory.documentCategory.findDatasetIdsByCategoryIds(categoryIds),
      ])

      // Step 4: union + dedupe. Set semantics handle the overlap case (a KB
      // grant and a category grant on the same KB produce duplicate IDs).
      const union = new Set<string>([...kbDatasets, ...categoryDatasets])

      // Step 5: soft cap. Warn + truncate if the fan-out is pathological.
      if (union.size > GRANT_DATASET_SOFT_CAP) {
        log.warn('resolveGrantedDatasetsForUser truncated: grant fan-out exceeds soft cap', {
          userId,
          tenantId,
          total: union.size,
          cap: GRANT_DATASET_SOFT_CAP,
        })
        return Array.from(union).slice(0, GRANT_DATASET_SOFT_CAP)
      }

      return Array.from(union)
    }
    ```
