    ```ts
    import type { Knex } from 'knex'

    /**
     * @description Phase 6 legacy role cleanup. Rewrites any lingering legacy
     * UserRole values (`'member'` → `'user'`, `'superadmin'` → `'super-admin'`)
     * and aborts if the table contains unknown role values.
     *
     * Scope notes (per Phase 6 CONTEXT.md amendment A-1):
     *   - The `users.role` column default is ALREADY `'user'` (see
     *     20260312000000_initial_schema.ts:33). No `ALTER COLUMN ... DEFAULT`
     *     statement is emitted here — it would be a no-op at best, a
     *     regression at worst.
     *   - The `'member'` default at 20260312000000_initial_schema.ts:93 is on
     *     `user_teams.role` (TeamRole.MEMBER) and is PRESERVED per D-04.
     *   - Seed file `00_sample_users.ts:295` also writes a TeamRole row and
     *     is NOT touched by this phase.
     *
     * @param {Knex} knex - Knex instance scoped to the migration transaction.
     */
    export async function up(knex: Knex): Promise<void> {
      // Canonical set of role values expected in `users.role`. Legacy values
      // `'member'` and `'superadmin'` are included here so the pre-check does
      // NOT treat them as unknown — they are rewritten in steps 2 and 3 below.
      const knownRoles = ['user', 'admin', 'super-admin', 'leader', 'member', 'superadmin']

      // Step 1 — Fail-fast pre-check. If the table contains any role value
      // outside the known set, abort before mutating anything. The error
      // message lists the distinct offending values so an operator can
      // reconcile them manually before re-running.
      const unknown = await knex('users')
        .whereNotIn('role', knownRoles)
        .distinct('role')
        .select('role')
      if (unknown.length > 0) {
        const values = unknown.map((r: { role: string }) => r.role).join(', ')
        throw new Error(
          `phase06_legacy_role_cleanup: users table contains unknown role values: ${values}. ` +
          `Reconcile these rows manually before re-running this migration.`,
        )
      }

      // Step 2 — Rewrite legacy `'member'` → `'user'`. Idempotent: the WHERE
      // clause matches zero rows on a re-run.
      await knex('users').where({ role: 'member' }).update({ role: 'user' })

      // Step 3 — Rewrite legacy `'superadmin'` (no hyphen) → `'super-admin'`
      // (canonical hyphenated form). Idempotent.
      await knex('users').where({ role: 'superadmin' }).update({ role: 'super-admin' })
    }

    /**
     * @description Intentional no-op. Per Phase 6 CONTEXT.md D-03 + amendment
     * A-1, the down migration does NOT reverse the UPDATEs — the role rename
     * is one-way data, and the Phase 3 `USE_ABILITY_ENGINE_V2=false` feature
     * flag is the real behavior rollback. Attempting to restore `'member'` /
     * `'superadmin'` values here would create a regression since the column
     * default and canonical enum keys no longer contain them.
     *
     * @param {Knex} _knex - Unused; retained for Knex API signature.
     */
    export async function down(_knex: Knex): Promise<void> {
      // No-op by design. See JSDoc above.
    }
    ```

    ```ts
    import { describe, it, expect } from 'vitest'
    import { withScratchDb, withScratchDbStoppingBefore } from './_helpers.js'

    /**
     * @description Integration tests for the phase06_legacy_role_cleanup
     * migration. Covers the pre-check abort path, the two conditional
     * UPDATEs, and idempotency. See phase-06 PLAN 6.1 and amendment A-1.
     */
    describe('phase06_legacy_role_cleanup migration', () => {
      const MIGRATION_SLUG = '_phase06_legacy_role_cleanup.ts'

      it('is a no-op on a clean migrated DB', async () => {
        await withScratchDb(async (knex) => {
          const rows = await knex('users').whereIn('role', ['member', 'superadmin']).select('id')
          expect(rows).toHaveLength(0)
        })
      })

      it("rewrites users.role='member' → 'user'", async () => {
        await withScratchDbStoppingBefore(MIGRATION_SLUG, async (knex) => {
          // Seed a user row carrying the legacy value BEFORE the cleanup runs
          await knex('users').insert({
            id: '00000000-0000-0000-0000-0000000000aa',
            email: 'legacy-member@test.local',
            role: 'member',
            // fill NOT NULL columns as needed — copy from backfill.test.ts seed shape
          })
          // Now run the migration explicitly
          await knex.migrate.up()
          const row = await knex('users')
            .where({ id: '00000000-0000-0000-0000-0000000000aa' })
            .first('role')
          expect(row?.role).toBe('user')
        })
      })

      it("rewrites users.role='superadmin' → 'super-admin'", async () => {
        await withScratchDbStoppingBefore(MIGRATION_SLUG, async (knex) => {
          await knex('users').insert({
            id: '00000000-0000-0000-0000-0000000000bb',
            email: 'legacy-sa@test.local',
            role: 'superadmin',
          })
          await knex.migrate.up()
          const row = await knex('users')
            .where({ id: '00000000-0000-0000-0000-0000000000bb' })
            .first('role')
          expect(row?.role).toBe('super-admin')
        })
      })

      it("aborts with a descriptive error when users table contains an unknown role", async () => {
        await withScratchDbStoppingBefore(MIGRATION_SLUG, async (knex) => {
          await knex('users').insert({
            id: '00000000-0000-0000-0000-0000000000cc',
            email: 'ghost@test.local',
            role: 'ghost',
          })
          await expect(knex.migrate.up()).rejects.toThrow(/ghost/)
          // Assert no mutation occurred on the seeded row
          const row = await knex('users')
            .where({ id: '00000000-0000-0000-0000-0000000000cc' })
            .first('role')
          expect(row?.role).toBe('ghost')
        })
      })

      it("is idempotent on re-run (second run is a no-op)", async () => {
        await withScratchDb(async (knex) => {
          // Running the migration file up() a second time manually by
          // re-importing its exports (see migrations.test.ts pattern) must

          await knex.migrate.latest()
          const rowsAfter = await knex('users').whereIn('role', ['member', 'superadmin']).count('* as c').first()
          expect(rowsAfter).toEqual(rowsBefore)
        })
      })
    ```
