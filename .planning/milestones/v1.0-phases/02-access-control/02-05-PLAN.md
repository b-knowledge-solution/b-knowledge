**Step 1: Add API functions**

In `fe/src/features/datasets/api/datasetApi.ts`, add:
```typescript
export async function updateDatasetPolicy(datasetId: string, policyRules: AbacPolicyRule[]): Promise<void> {
  await api.put(`/knowledge-base/${datasetId}/policy`, { policy_rules: policyRules })
}
```

In `fe/src/features/datasets/api/datasetQueries.ts`, add:
```typescript
export function useUpdateDatasetPolicy() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: ({ datasetId, policyRules }: { datasetId: string; policyRules: AbacPolicyRule[] }) =>
      updateDatasetPolicy(datasetId, policyRules),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.datasets.all })
      toast.success(t('accessControl.policy.saveSuccess'))
    },
    onError: () => {
      toast.error(t('accessControl.policy.saveError'))
    },
  })
}
```

**Step 1: Add API functions for role management**

In `fe/src/features/users/api/userApi.ts` (create if it doesn't exist), add:
```typescript
export async function updateUserRole(userId: string, role: string): Promise<void> {
  await api.put(`/users/${userId}/role`, { role })
}

export async function getOrgMembers(): Promise<OrgMember[]> {
  const res = await api.get('/users')
  return res.data
}
```

In `fe/src/features/users/api/userQueries.ts` (create if it doesn't exist), add:
```typescript
export function useOrgMembers() {
  return useQuery({
    queryKey: queryKeys.users.all,
    queryFn: getOrgMembers,
  })
}

export function useUpdateUserRole() {
  const queryClient = useQueryClient()
  const { t } = useTranslation()
  return useMutation({
    mutationFn: ({ userId, role }: { userId: string; role: string }) => updateUserRole(userId, role),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.users.all })
      toast.success(t('accessControl.roleManagement.updateSuccess'))
    },
    onError: () => {
      toast.error(t('accessControl.roleManagement.updateError'))
    },
  })
}
```
