   ```js

   export function transformSource(source, filename) {
     const project = new Project({ useInMemoryFileSystem: true })
     const file = project.createSourceFile(filename, source, { overwrite: true })
     // ... transforms ...
     return file.getFullText()
   }
   ```

4. **Pattern B — `const isAdmin = user?.role === UserRole.X` rewrite (SINGLE-ROLE ONLY):**
   - Find `VariableDeclaration` nodes named `isAdmin` whose initializer is a SINGLE `BinaryExpression` matching exactly `user?.role === UserRole.ADMIN` (or the string-literal form `user?.role === 'admin'`).
   - **Multi-role OR-chains are NOT auto-rewritten** (semantic gap — there is no single catalog key meaning "admin OR leader" because catalog keys are capability-scoped, not role-scoped). When the initializer is a `BinaryExpression` with operator `||` (e.g. `user?.role === ADMIN || user?.role === LEADER`), leave the declaration unchanged and prepend the comment `// TODO(perm-codemod): multi-role chain — manual migration required (split into useHasPermission calls per capability)`. This is enforced as fixture **08** below.
   - For single-role matches, look up the file in the **`FILE_KEY_MAP` table below** (literal `Record<string, string>` baked into the lib, NOT computed from RESEARCH at runtime). If the file is in the map, replace the initializer with `useHasPermission(PERMISSION_KEYS.<KEY>)`. If the file is NOT in the map, leave the declaration unchanged and prepend `// TODO(perm-codemod): pick a PERMISSION_KEYS.<KEY> for this gate`.

```ts
import { describe, it, expect } from 'vitest'
import { readFileSync, readdirSync, statSync } from 'node:fs'
import { join, dirname } from 'node:path'
import { fileURLToPath } from 'node:url'
import { transformSource } from '../codemod-permissions.lib.mjs'

describe('codemod-permissions', () => {
  for (const fixture of readdirSync(FIXTURES_DIR)) {
    if (!statSync(join(FIXTURES_DIR, fixture)).isDirectory()) continue
    it(`${fixture} matches expected output`, () => {
      const dir = join(FIXTURES_DIR, fixture)
      const inputFile = readdirSync(dir).find(f => f.startsWith('input.'))
      const expectedFile = readdirSync(dir).find(f => f.startsWith('expected.'))
      const input = readFileSync(join(dir, inputFile), 'utf8')
      const expected = readFileSync(join(dir, expectedFile), 'utf8')
      // Use a representative path so skip-list and heuristic mapping behave realistically
      const filename = `fe/src/features/datasets/components/${inputFile}`
      const actual = transformSource(input, filename)
      expect(actual.trim()).toBe(expected.trim())
    })
  }
})
```
